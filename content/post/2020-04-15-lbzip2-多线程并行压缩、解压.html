安装

wget https://launchpad.net/ubuntu/+archive/primary/+sourcefiles/lbzip2/2.5-2/lbzip2_2.5.orig.tar.bz2

tar -xvf lbzip2_2.5.orig.tar.bz2
cd lbzip2-2.5

sudo ./configure --prefix=/usr
sudo make -j
sudo make install
