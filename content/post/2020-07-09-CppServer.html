安装

apt-get install uuid-dev

pip3 install gil

git clone https://github.com/chronoxor/CppServer.git
cd CppServer
gil update

cd modules
rm -rf asio
pc git clone git@github.com:chriskohlhoff/asio.git

编译

cd build
./unix.sh
